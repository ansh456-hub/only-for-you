const express = require("express");
const path = require("path");

const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.json());
app.use(express.static(path.join(__dirname, "public")));

app.post("/chat", (req, res) => {
    const userMessage = req.body.message.toLowerCase();

    let reply = "";

    if (userMessage.includes("hello")) {
        reply = "Hello my love 💕";
    } 
    else if (userMessage.includes("love")) {
        reply = "I love you ❤️ – it's from your Krishna 💌";
    } 
    else if (userMessage.includes("valentine")) {
        reply = "Happy Valentine’s Day my forever 💘";
    } 
    else {
        reply = "No matter what you say... I still love you ❤️ – from your Krishna";
    }

    res.json({ reply });
});

app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
